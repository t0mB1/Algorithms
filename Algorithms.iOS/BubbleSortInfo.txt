Worst: O(n²)
Average: O(n²)
Bubble sort is the simplest sorting algorithm that operates by repeatedly swapping adjacent items if they are in the wrong order.

C# Implementation:

using System;
class Algorithm
{
    static int[] BubbleSort(int searchItem, int[] array)
    {
        int n = array.Length;
        for(int i = 0; i < n - 1; i++)
        {
            for(int j = 0; j < n - 1; j++)
            {
                if(array[j] > array[j + 1])
                {
                    int temp = array[j];
                    array[j] = array[j + 1]
                    array[j + 1] = array[j]
                }
            }
        }
        return array[];
    }
}
